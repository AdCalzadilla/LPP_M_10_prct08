def pop()
  freeNode = @head.value
  if @head.next == nil
    return freeNode
  end
  @head = @head.next
  @head.prev = nil
  freeNode
end

def print_reverse
  current = @head
  while current.next != nil
    current = current.next
  end
  while current.prev != nil
    puts current.value
    current = current.prev     
  end
  puts current.value
end
